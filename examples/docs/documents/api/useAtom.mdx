## useAtom

- You cannot expect something like this to work:

```jsx
import { useAtom } from 'react-mol'
function App () {
  const atom = useAtom()
  return (
    <Instanced>
      <meshPhongMaterial/>
      ...
```

- Do this instead:

```jsx
function Foo () {
  const atom = useAtom()
}

function App () {
  return (
    <Instanced>
      <Foo>
      ...
```
